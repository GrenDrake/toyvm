.export start

hello_msg:
    .string "Hello world!"

start:
    pushw hello_msg
    saystr

    pushb    10
loop:
    stkdup
    saynum
    pushb    1
    sub
    stkdup
    pushw    loop
    jnz

    pushb    0
    readw
    saynum
    pushb    0
    reads
    saynum
    pushb    0
    readb
    saynum

    exit
